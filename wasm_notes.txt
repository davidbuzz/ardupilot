#/bin/bash
# shell script and notes to compile arduplane sitl with emscripten to .wasm and javascript.
# wasm_notes.txt

# if u don't have these, install them..
#sudo apt install cmake git python3 

#colors:
RED='\033[0;31m'
NC='\033[0m' # No Color

echo -e "${RED}getting emscrypten via its sdk:${NC}"

cd ~
if [ ! -d "./emsdk" ] 
then
    git clone https://github.com/emscripten-core/emsdk.git
    cd emsdk
    ./emsdk install latest
    ./emsdk activate latest

    echo 'source "~/emsdk_env.sh"' >> $HOME/.bash_profile
    source "~/emsdk/emsdk_env.sh"
    cd ~
fi

which emcc
which em++

echo -e "${RED}setting emcc/em++ as the compiler to use:${NC}"

export CC=emcc
export CXX=em++
export EMCC_DEBUG=1

echo -e "${RED}configuring arduplane for emscriptem/wasm:${NC}"


if [ ! -d ~/ardupilot ]
then
  echo -e "${RED} Sorry, this script assumes ardupilot sources/git repo is at ~/ardupilot , giving up, change script to suite your setup.${NC}"
  exit
fi

cd ~/ardupilot

# hack to empty SHLIB_MARKER and STLIB_MARKER as '-Wl,-Bdynamic' and '-Wl,-Bstatic' are not supported by emscrypten/clang, and i don't know why its using the waflib 'gcc' and 'gxx' tools?
cp gcc.py modules/waf/waflib/Tools/gcc.py
cp gxx.py modules/waf/waflib/Tools/gxx.py


./waf configure board=sitl --disable-scripting -v

echo -e "${RED}building arduplane for emscriptem/wasm${NC}"

./waf plane -v -v 


echo -e "${RED}setting up web-server for arduplane/emscriptem/wasm:${NC}"

# web page that loads arduplane.js and arduplane.wasm
cp ./shell_minimal.html build/sitl/bin/index.html

# target file is a .js but doesn't get the right extension, so we copy it to .js:
cp build/sitl/bin/arduplane build/sitl/bin/arduplane.js

echo -e "goto: ${RED}http://localhost:8000/${NC} or ctrl-c to exit web server"
( cd build/sitl/bin/ ; emrun --port 8000 . )

exit
